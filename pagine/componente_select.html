Componente select

La select ci permette di effettuare una scelta all'interno di un menù a tendina

Codice di esempio per inserire una select

						
import React, { useState } from 'react';

function App() {

    //DICHIARAZIONE DELLO useState dove salvo il valore della scelta selezionata
    const [valoreSelect, setValoreSelect] = useState('');

    //FUNZIONE AGGIORNA (richiamata automaticamente)
    const aggiorna = (event) => {
        setValoreSelect(event.target.value);
        alert(`Hai selezionato: ${event.target.value}`);
    };

    //RETURN --> Stampa a video
    return (
        <div>
            <select value={valoreSelect} onChange={aggiorna}>
                <option value="">Scegli un opzione</option>
                <option value="option1">Opzione 1</option>
                <option value="option2">Opzione 2</option>
                <option value="option3">Opzione 3</option>
            </select>
        </div>
    );
}

export default App;
